example ping pong bot

```js
const Discord = require("nerdcord")

const Client = new Discord.client()

Client.login("token")

Client.on("ready", () => {
    console.log("ready")
})

Client.on("message", async (message) => {
    if (message.content === "?ping") {
        Client.createMessage("pong", message.channel_id, "token")
    }
})```
